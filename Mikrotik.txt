## ROUNTER 1##
/ip address add address=172.16.200.1/16 interface=ether1
/ip address add address=192.200.10.1/24 interface=ether2
/ip route add gateway=172.16.0.1
/ip dns set servers=8.8.8.8
/ip firewall nat add chain=srcnat out-interface=ether1 action=masquerade
/ip proxy set enable=yes port=3128 cache-administrator=admin@sekolah.sch.id
/ip firewall nat add chain=dstnat protocol=tcp dst-port=80 action=redirect to-ports=3128 in-interface=ether2
/ip proxy access add dst-host=www.linux.org action=deny
/routing ospf interface add interface=ether1
/routing ospf network add network=172.16.0.0/16 area=backbone
/routing ospf network add network=192.200.10.0/24 area=backbone
